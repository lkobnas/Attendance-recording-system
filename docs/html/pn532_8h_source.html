162 #define PN532_ERROR_NONE (0x00)
163 // Time Out, the target has not answered
164 #define PN532_ERROR_TIMEOUT (0x01)
165 // A CRC error has been detected by the CIU
166 #define PN532_ERROR_CRC (0x02)
167 // A Parity error has been detected by the CIU
168 #define PN532_ERROR_PARITY (0x03)
169 // During an anti-collision/select operation (ISO/IEC14443-3 Type A and
170 // ISO/IEC18092 106 kbps passive mode), an erroneous Bit Count has been
171 // detected
172 #define PN532_ERROR_COLLISION_BITCOUNT (0x04)
173 // Framing error during MIFARE operation
174 #define PN532_ERROR_MIFARE_FRAMING (0x05)
175 // An abnormal bit-collision has been detected during bit wise
176 // anti-collision at 106 kbps
177 #define PN532_ERROR_COLLISION_BITCOLLISION (0x06)
178 // Communication buffer size insufficien
179 #define PN532_ERROR_NOBUFS (0x07)
180 // RF Buffer overflow has been detected by the CI
181 #define PN532_ERROR_RFNOBUFS (0x09)
182 // In active communication mode, the RF field has not been switched on
183 // in time by the counterpart (as defined in NFCIP-1 standard
184 #define PN532_ERROR_ACTIVE_TOOSLOW (0x0a)
185 // RF Protocol error
186 #define PN532_ERROR_RFPROTO (0x0b)
187 // Temperature error: the internal temperature sensor has detected
188 // overheating, and therefore has automatically switched off the
189 // antenna drivers
190 #define PN532_ERROR_TOOHOT (0x0d)
191 // Internal buffer overflow
192 #define PN532_ERROR_INTERNAL_NOBUFS (0x0e)
193 // Invalid parameter (range, format...)
194 #define PN532_ERROR_INVAL (0x10)
195 // DEP Protocol: The PN533 configured in target mode does not support
196 // the command received from the initiator (the command received is not
197 // one of the following: ATR_REQ, WUP_REQ, PSL_REQ, DEP_REQ, DSL_REQ,
198 // RLS_REQ)
199 #define PN532_ERROR_DEP_INVALID_COMMAND (0x12)
200 // DEP Protocol, MIFARE or ISO/IEC14443-4: The data format does not
201 // match to the specification. Depending on the RF protocol used, it
202 // can be: Bad length of RF received frame, Incorrect value of PCB or
203 // PFB, Invalid or unexpected RF received frame, NAD or DID incoherence.
204 #define PN532_ERROR_DEP_BADDATA (0x13)
205 // MIFARE: Authentication error
206 #define PN532_ERROR_MIFARE_AUTH (0x14)
207 // Target or Initiator does not support NFC Secur
208 #define PN532_ERROR_NOSECURE (0x18)
209 // I2C bus line is Busy. A TDA transaction is on going
210 #define PN532_ERROR_I2CBUSY (0x19)

212 #define PN532_ERROR_UIDCHECKSUM (0x23)
